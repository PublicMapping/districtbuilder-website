34  @register.filter 
35  @stringfilter 
36 -def truncate(value, length): 

37      """ 
38      This filter reduces the given string value to length  
39      and appends an ellipsis to the end. If the string is 
40      not of the given length, it is returned as-is 
41      Parameters: 
42          value - A string value 
43          length - The length of the string desired (ellipsis included) 
44      """ 
45      try: 
46          length = int(length) 
47          if len(value) > length: 
48              short = format('%s%s' % (value[:length - 3], '...')) 
49              return short 
50          return value 
51      except: 
52          return value

66           "seventeen", "eighteen", "nineteen", "twenty"][value]

70  @register.filter 
71  @stringfilter 
72 -def district_number(value): 

73      """ 
74      This filter removes any extra words from a district name. 
75      E.g., "District 10" becomes "10". 
76       
77      Special case: "Unassigned" becomes a Theta symbol 
78   
79      Parameters: 
80          a full district name: 
81      Returns: 
82          a number 
83      """ 
84      try: 
85          if value == "Unassigned": 
86              return mark_safe('&#216;') 
87          p = re.compile('(?P<member>(\w+\s+)+)(?P<digits>\d+)') 
88          m = p.match(value) 
89          if m != None: 
90              return m.group('digits')  
91      except: 
92          return value
