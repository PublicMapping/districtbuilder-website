  53          a message to return indicating any changes. 
  54      """ 
  55      name = kwargs[unique_id_field] 
  56      object_name = '%s %s' % (a_model.__name__, name) 
  57      # Get the model if it exists 
  58      try: 
  59          id_args = { unique_id_field: name } 
  60          current_object = a_model.objects.get(**id_args) 
  61      # If it doesn't exist, just save it and return 
  62      except ObjectDoesNotExist: 
  63          new = a_model(**kwargs) 
  64          new.save() 
  65          return new, True, False, '%s created' % object_name 
  66      # If it exists, track any changes and overwrite if requested 
  67      different = [] 
  68      changed = False 
  69      message = '%s matches database - no changes%s' 
  70      for key in kwargs: 
  71          current_value = current_object.__getattribute__(key) 
  72          config_value = kwargs[key] 
  73          if not (isinstance(current_value, types.StringTypes)) and not (isinstance(current_value, Model)):

  78                  changed = True
